# Install MongoDB, Ruby, Gem and etc
```bash
brew install ruby
gem install mongo
gem install bundler
bundle install
```
```ruby - connect.rb
require 'rubygems'
require 'mongo'
$client = Mongo::Client.new([ '127.0.0.1:27017' ], :database => 'tutorial')
Mongo::Logger.logger.level = ::Logger::ERROR
$users = $client[:users]
puts 'connected!'
```
